if (todo) {
const todoItem = $('<li class="collection-item">')
.text(todo)

todoItems.append(todoItem);

todoInput.val(''); // clear input
}
}
});

// Remove Todo Functionality

$(this)
.parents('.collection-item')
.remove();
});
});`
