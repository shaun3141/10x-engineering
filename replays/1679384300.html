// Define necessary elements
var ul = $("ul.collection");
var form = $("form");
var newTodoInput = $("#newTodo");

// Listen for form submission
form.submit(function (event) {
event.preventDefault();
var newTodo = newTodoInput.val();
if (newTodo !== "") {
addTodoItem(newTodo);
}
});

// Function to add new Todo item to list
function addTodoItem(itemText) {
ul.append('<li class="collection-item">' + itemText + "</li>");
newTodoInput.val("");
}
});

$(document).ready(function () {
var todoList = $("#todo-list");
var todoForm = $("#todo-form");
var todoInput = $("#todo");
var todos = [];

todoForm.submit(function (event) {
event.preventDefault();
var newTodo = todoInput.val();
if (newTodo.length > 0) {
todos.push(newTodo);
updateTodoList();
todoInput.val("");
}
});

function updateTodoList() {
todoList.empty();
for (var i = todos.length - 1; i >= 0; i--) {
var todoItem = $("<li>").addClass("collection-item").text(todos[i]);
todoList.append(todoItem);
}
}
});
