from typing import List
import ast


def custom_calculation(expression: str, numbers: List[float]) -> float:
# create a function from the input expression and evaluate it
f = ast.parse(expression.strip(), mode='eval')
code = compile(f, '<string>', 'eval')
result = eval(code, {'__builtins__': None}, {'x': numbers})
